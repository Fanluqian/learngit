#include <stdio.h>
#include <stdlib.h>

/* run this program using the console pauser or add your own getch, system("pause") or input loop */

int main(int argc, char *argv[]) {
	int n,i,j,k,a;
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		for(j=1;j<=n+i-1;j++){
		if(j<=n-i){
			printf("  ");
		}
		else if(j>n-i && j<n){
			for(a=n;a>n-i+1;a--,j++)
			printf("%-2d",a);
		}
		else if(j>=n && j<=n+i-1){
			for(a=n-i+1;a<=n;a++,j++)
			printf("%-2d",a);
		}
	} printf("\n");
	}
	return 0;
	}
